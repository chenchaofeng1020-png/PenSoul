# 产品资料维护功能设计方案

## 一、项目背景与目标

### 1.1 现状分析
当前产品鸭平台已具备竞品管理和内容规划功能，但产品基础信息管理相对薄弱，主要存在以下问题：
- 产品信息仅包含基础字段（名称、描述、Logo、官网）
- 缺乏产品卖点、特性等营销关键信息管理
- 产品资料与内容规划、竞品分析功能割裂
- 无法为内容创作提供标准化的产品信息支撑

### 1.2 设计目标
构建一个**独立且专注**的产品资料管理模块，实现：
- **专注信息管理**：只做产品基础信息、卖点、特性的维护
- **融入现有架构**：作为当前产品空间的一个功能模块
- **与内容规划分离**：产品资料管理和内容创作是两个独立板块
- **支持现有模型**：基于当前的product数据结构进行扩展

## 二、系统架构集成方案


- **数据库**：SQLite 3（扩展现有数据库结构）
- **文件存储**：本地文件系统（复用现有上传机制）

### 5.1 控制器架构
```php
// backend/src/Controllers/ProductDataController.php
namespace App\Controllers;

use App\Models\Product;
use App\Models\ProductSellingPoint;
use App\Models\ProductFeature;
use App\Models\ProductVersion;
use App\Utils\Response;
use App\Middleware\AuthMiddleware;

class ProductDataController
{
    private $productModel;
    private $sellingPointModel;
    private $featureModel;
    private $versionModel;

    public function __construct()
    {
        $this->productModel = new Product();
        $this->sellingPointModel = new ProductSellingPoint();
        $this->featureModel = new ProductFeature();
        $this->versionModel = new ProductVersion();
    }

    /**
     * 获取产品资料列表（包含统计信息）
     */
    public function index()
    {
        try {
            $userId = AuthMiddleware::getCurrentUserId();
            if (!$userId) {
                return Response::error('未授权访问', 401);
            }

            // 获取查询参数
            $page = isset($_GET['page']) ? max(1, intval($_GET['page'])) : 1;
            $limit = isset($_GET['limit']) ? max(1, min(50, intval($_GET['limit']))) : 10;
            $search = isset($_GET['search']) ? trim($_GET['search']) : '';

            // 获取用户产品列表（扩展字段）
            $result = $this->productModel->getUserProductsWithStats($userId, $page, $limit, $search);

            if ($result === false) {
                return Response::error('获取产品资料列表失败');
            }

            return Response::success($result, '获取产品资料列表成功');
        } catch (\Exception $e) {
            error_log('Get products data error: ' . $e->getMessage());
            return Response::error('服务器内部错误');
        }
    }

    /**
     * 获取完整的产品资料详情
     */
    public function show($id)
    {
        try {
            $userId = AuthMiddleware::getCurrentUserId();
            if (!$userId) {
                return Response::error('未授权访问', 401);
            }

            // 验证产品权限
            if (!$this->productModel->checkUserAccess($id, $userId)) {
                return Response::error('无权限访问该产品', 403);
            }

            // 获取产品基础信息
            $product = $this->productModel->findById($id);
            if (!$product) {
                return Response::error('产品不存在', 404);
            }

            // 获取关联数据
            $product['selling_points'] = $this->sellingPointModel->getByProductId($id);
            $product['features'] = $this->featureModel->getByProductId($id);
            $product['versions'] = $this->versionModel->getByProductId($id);
            $product['stats'] = $this->productModel->getProductDataStats($id);

            return Response::success($product, '获取产品资料详情成功');
        } catch (\Exception $e) {
            error_log('Get product data detail error: ' . $e->getMessage());
            return Response::error('服务器内部错误');
        }
    }

    /**
     * 更新产品资料（包含完善度计算）
     */
    public function update($id)
    {
        try {
            $userId = AuthMiddleware::getCurrentUserId();
            if (!$userId) {
                return Response::error('未授权访问', 401);
            }

            // 验证权限
            if (!$this->productModel->checkUserAccess($id, $userId)) {
                return Response::error('无权限访问该产品', 403);
            }

            // 获取请求数据
            $input = json_decode(file_get_contents('php://input'), true);
            if (!$input) {
                return Response::error('无效的请求数据', 400);
            }

            // 数据验证
            $validation = $this->validateProductData($input);
            if ($validation !== true) {
                return Response::error('数据验证失败', 400, $validation);
            }

            // 计算完善度
            $completionData = $this->calculateCompletionRate($input);
            
            // 准备更新数据
            $updateData = array_merge($input, [
                'completion_rate' => $completionData['rate'],
                'updated_at' => date('Y-m-d H:i:s')
            ]);

            // 更新产品资料
            $product = $this->productModel->updateProductData($id, $updateData);
            if (!$product) {
                return Response::error('更新产品资料失败');
            }

            // 记录版本历史（如果有重要变更）
            if ($completionData['has_major_change']) {
                $this->versionModel->create([
                    'product_id' => $id,
                    'version' => $product['current_version'] ?? 'v1.0.0',
                    'changelog' => '产品资料更新',
                    'created_by' => $userId
                ]);
            }

            return Response::success($product, '更新产品资料成功');
        } catch (\Exception $e) {
            error_log('Update product data error: ' . $e->getMessage());
            return Response::error('服务器内部错误');
        }
    }

    /**
     * 计算产品资料完善度
     */
    private function calculateCompletionRate($data)
    {
        $totalFields = 8; // 总字段数
        $filledFields = 0;
        $hasMajorChange = false;

        $requiredFields = ['name', 'description', 'slogan', 'target_user', 'pricing_info', 'current_version', 'website_url', 'logo'];
        
        foreach ($requiredFields as $field) {
            if (!empty($data[$field])) {
                $filledFields++;
                if (in_array($field, ['slogan', 'target_user', 'pricing_info'])) {
                    $hasMajorChange = true;
                }
            }
        }

        return [
            'rate' => round(($filledFields / $totalFields) * 100),
            'has_major_change' => $hasMajorChange
        ];
    }
}
```

```php
// backend/src/Models/ProductData.php
namespace App\Models;

use App\Config\Database;
use PDO;

class ProductData extends Product
{
    /**
     * 获取包含扩展统计信息的产品列表
     */
    public function getUserProductsWithStats($userId, $page = 1, $limit = 10, $search = '')
    {
        try {
            $offset = ($page - 1) * $limit;
            
            // 基础查询（包含扩展字段）
            $sql = "SELECT 
                        p.*,
                        u.username as creator_username,
                        COUNT(DISTINCT sp.id) as selling_points_count,
                        COUNT(DISTINCT f.id) as features_count,
                        COUNT(DISTINCT v.id) as versions_count,
                        (SELECT COUNT(*) FROM competitors c WHERE c.product_id = p.id AND c.deleted_at IS NULL) as competitor_count
                    FROM products p 
                    INNER JOIN user_products up ON p.id = up.product_id 
                    LEFT JOIN users u ON p.user_id = u.id 
                    LEFT JOIN product_selling_points sp ON p.id = sp.product_id AND sp.is_active = 1
                    LEFT JOIN product_features f ON p.id = f.product_id 
                    LEFT JOIN product_versions v ON p.id = v.product_id 
                    WHERE up.user_id = :user_id AND p.deleted_at IS NULL 
                    GROUP BY p.id
                    ORDER BY p.created_at DESC 
                    LIMIT :limit OFFSET :offset";
            
            $stmt = $this->db->prepare($sql);
            $stmt->bindParam(':user_id', $userId);
            $stmt->bindParam(':limit', $limit, PDO::PARAM_INT);
            $stmt->bindParam(':offset', $offset, PDO::PARAM_INT);
            $stmt->execute();
            
            $products = $stmt->fetchAll(PDO::FETCH_ASSOC);
            
            // 获取总数
            $countSql = "SELECT COUNT(DISTINCT p.id) as total 
                        FROM products p 
                        INNER JOIN user_products up ON p.id = up.product_id 
                        WHERE up.user_id = :user_id AND p.deleted_at IS NULL";
            
            $countStmt = $this->db->prepare($countSql);
            $countStmt->bindParam(':user_id', $userId);
            $countStmt->execute();
            $total = $countStmt->fetch(PDO::FETCH_ASSOC)['total'];
            
            return [
                'products' => $products,
                'pagination' => [
                    'current_page' => $page,
                    'per_page' => $limit,
                    'total' => $total,
                    'total_pages' => ceil($total / $limit)
                ]
            ];
        } catch (PDOException $e) {
            error_log("Get user products with stats error: " . $e->getMessage());
            return false;
        }
    }

    /**
     * 更新产品资料（包含扩展字段）
     */
    public function updateProductData($id, $data)
    {
        try {
            $fields = [];
            $params = [':id' => $id];
            
            $allowedFields = [
                'name', 'description', 'logo', 'website_url', 'slogan', 
                'target_user', 'pricing_info', 'current_version', 'completion_rate'
            ];
            
            foreach ($allowedFields as $field) {
                if (isset($data[$field])) {
                    $fields[] = "{$field} = :{$field}";
                    $params[":{$field}"] = $data[$field];
                }
            }
            
            if (empty($fields)) {
                return false;
            }
            
            $fields[] = "updated_at = :updated_at";
            $params[":updated_at"] = date('Y-m-d H:i:s');
            
            $sql = "UPDATE products SET " . implode(', ', $fields) . " WHERE id = :id";
            
            $stmt = $this->db->prepare($sql);
            foreach ($params as $key => $value) {
                $stmt->bindValue($key, $value);
            }
            
            if ($stmt->execute()) {
                return $this->findById($id);
            }
            
            return false;
        } catch (PDOException $e) {
            error_log("Update product data error: " . $e->getMessage());
            return false;
        }
    }
}
```
